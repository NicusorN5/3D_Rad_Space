cmake_minimum_required(VERSION 3.21)

project("3DRadSpace.Logging")

add_library(3DRadSpace.Logging SHARED 
	"Exception.hpp"
	"Exception.cpp"
	"Logging.hpp"
	"Message.hpp"
	"Message.cpp"
	"Message.h"
	"ServiceBadCastException.hpp"
	"Warning.hpp"
	"Warning.cpp"
	"Warning.h"
)

if (CMAKE_VERSION VERSION_GREATER 3.12)
  set_property(TARGET 3DRadSpace.Logging PROPERTY CXX_STANDARD 23)
endif()

if(MSVC)
	#Disable warning 4251 : 'type' : class 'type1' needs to have dll-interface to be used by clients of class 'type2'
	target_compile_options(3DRadSpace.Logging PRIVATE "/wd4251")
	#Fix C4530
	target_compile_options(3DRadSpace.Logging PRIVATE "/EHsc")
	# Create separate pdb debug info files
	add_compile_options(3DRadSpace.Logging PRIVATE "/Zi")
	# Multiprocessor build
	add_compile_options(3DRadSpace.Logging PRIVATE "/MP")

	#enable address sanitizer if using a debug configuration
	if(CMAKE_BUILD_TYPE STREQUAL "Debug")
		target_compile_options(3DRadSpace.Logging PRIVATE "/fsanitize=address")
	endif()

	target_compile_definitions(3DRadSpace.Logging PRIVATE [[E3DRSP_LOGGING_EXPORT=__declspec(dllexport)]])
endif()

if(WIN32)
	target_compile_definitions(3DRadSpace.Logging PRIVATE _WINDLL=1)
endif()

#
#Dependencies 
#
target_link_libraries(3DRadSpace.Logging PRIVATE 3DRadSpace.Core)
target_include_directories(3DRadSpace.Logging PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/../)
